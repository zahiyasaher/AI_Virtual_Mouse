# virtual_mouse_all_in_one.py
import cv2
import mediapipe as mp
import time
import math
import numpy as np
import autopy
import pyautogui

# ---------- Hand Detector ----------
class handDetector():
    def __init__(self, mode=False, maxHands=1, modelComplexity=1, detectionCon=0.5, trackCon=0.5):
        self.mode = mode
        self.maxHands = maxHands
        self.modelComplex = modelComplexity
        self.detectionCon = detectionCon
        self.trackCon = trackCon
        self.mpHands = mp.solutions.hands
        self.hands = self.mpHands.Hands(self.mode, self.maxHands, self.modelComplex,
                                        self.detectionCon, self.trackCon)
        self.mpDraw = mp.solutions.drawing_utils
        self.tipIds = [4, 8, 12, 16, 20]

    def findHands(self, img, draw=True):
        imgRGB = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
        self.results = self.hands.process(imgRGB)

        if self.results.multi_hand_landmarks:
            for handLms in self.results.multi_hand_landmarks:
                if draw:
                    self.mpDraw.draw_landmarks(img, handLms, self.mpHands.HAND_CONNECTIONS)
        return img

    def findPosition(self, img, handNo=0, draw=True):
        xList, yList = [], []
        bbox = []
        self.lmList = []

        if self.results.multi_hand_landmarks:
            myHand = self.results.multi_hand_landmarks[handNo]
            h, w, _ = img.shape
            for id, lm in enumerate(myHand.landmark):
                cx, cy = int(lm.x * w), int(lm.y * h)
                xList.append(cx)
                yList.append(cy)
                self.lmList.append([id, cx, cy])

                if draw:
                    cv2.circle(img, (cx, cy), 5, (255, 0, 255), cv2.FILLED)

            xmin, xmax = min(xList), max(xList)
            ymin, ymax = min(yList), max(yList)
            bbox = xmin, ymin, xmax, ymax

            if draw:
                cv2.rectangle(img, (xmin - 20, ymin - 20),
                              (xmax + 20, ymax + 20), (0, 255, 0), 2)

        return self.lmList, bbox

    def fingersUp(self):
        fingers = [0,0,0,0,0]
        if len(self.lmList) == 0:
            return fingers

        # Thumb
        if self.lmList[self.tipIds[0]][1] > self.lmList[self.tipIds[0]-1][1]:
            fingers[0] = 1

        # Other fingers
        for i in range(1,5):
            if self.lmList[self.tipIds[i]][2] < self.lmList[self.tipIds[i]-2][2]:
                fingers[i] = 1

        return fingers

    def findDistance(self, p1, p2, img=None, draw=True, r=8, t=2):
        x1, y1 = self.lmList[p1][1:]
        x2, y2 = self.lmList[p2][1:]
        cx, cy = (x1 + x2)//2, (y1 + y2)//2
        if draw and img is not None:
            cv2.line(img, (x1,y1), (x2,y2), (255,0,255), t)
            cv2.circle(img,(x1,y1), r, (255,0,255), cv2.FILLED)
            cv2.circle(img,(x2,y2), r, (255,0,255), cv2.FILLED)
            cv2.circle(img,(cx,cy), r, (0,0,255), cv2.FILLED)
        length = math.hypot(x2-x1, y2-y1)
        return length, img, [x1,y1,x2,y2,cx,cy]


# ---------- Main Program ----------
def main():
    wCam, hCam = 1280, 720
    frameR = 100
    smoothening = 7

    pTime = 0
    plocX, plocY = 0, 0

    drag_active = False

    cap = cv2.VideoCapture(0)
    cap.set(3, wCam)
    cap.set(4, hCam)
    detector = handDetector(maxHands=1)

    wScr, hScr = autopy.screen.size()

    print("Virtual Mouse Started -- Press 'q' to quit")
    print("Controls:")
    print("- Index finger only: Move cursor")
    print("- Index + Middle close together: Click")
    print("- Thumb + Index close: Right click")
    print("- Index + Middle UP (fingers spread): Scroll UP")
    print("- Index + Middle DOWN (fingers spread): Scroll DOWN")
    print("- Scroll speed = horizontal finger gap")

    while True:
        success, img = cap.read()
        if not success:
            continue

        img = detector.findHands(img)
        lmList, _ = detector.findPosition(img)
        fingers = [0,0,0,0,0]

        if lmList:
            fingers = detector.fingersUp()

            xIndex, yIndex = lmList[8][1:]
            xMiddle, yMiddle = lmList[12][1:]
            xThumb,  yThumb  = lmList[4][1:]

            # Get PIP joint positions (middle knuckles) for orientation detection
            xIndex_pip, yIndex_pip = lmList[6][1:]    # Index finger PIP joint
            xMiddle_pip, yMiddle_pip = lmList[10][1:]  # Middle finger PIP joint

            cv2.rectangle(img, (frameR, frameR),
                          (wCam - frameR, hCam - frameR),
                          (255,0,255), 2)

            # Calculate distance between index and middle fingers
            length_fingers = math.hypot(xMiddle - xIndex, yMiddle - yIndex)
            
            # Check if fingers are EXTENDED (regardless of up/down direction)
            # Calculate distance from base to tip for each finger
            index_length = math.hypot(xIndex - lmList[5][1], yIndex - lmList[5][2])
            middle_length = math.hypot(xMiddle - lmList[9][1], yMiddle - lmList[9][2])
            
            # Fingers are extended if tip is far from base (threshold: 100 pixels)
            index_extended = index_length > 100
            middle_extended = middle_length > 100
            
            # ------------------- SCROLL MODE vs CLICK (check horizontal distance first) -------------------
            # Use extension check instead of fingersUp() to work in any direction
            if index_extended and middle_extended:
                horiz_dist = abs(xMiddle - xIndex)
                
                # CLICK: fingers close together (small horizontal distance)
                if horiz_dist < 40:
                    if drag_active:
                        autopy.mouse.toggle(autopy.mouse.Button.LEFT, False)
                        drag_active = False
                    
                    autopy.mouse.click()
                    cv2.putText(img, 'CLICK', (wCam//2 - 50, 100),
                               cv2.FONT_HERSHEY_PLAIN, 2, (255, 255, 0), 3)
                    time.sleep(0.20)
                
                # SCROLL: fingers spread apart (large horizontal distance)
                elif horiz_dist > 50:
                    # sensitivity: smaller divisor -> faster scroll growth
                    speed = max(1, int(horiz_dist / 6))
                    speed = min(speed, 80)  # hard cap to avoid huge jumps

                    # DEBUG INFO
                    cv2.putText(img, f'Index: tip={yIndex} pip={yIndex_pip} diff={yIndex-yIndex_pip}', (20, 100),
                                cv2.FONT_HERSHEY_PLAIN, 1.3, (255, 255, 255), 2)
                    cv2.putText(img, f'Middle: tip={yMiddle} pip={yMiddle_pip} diff={yMiddle-yMiddle_pip}', (20, 130),
                                cv2.FONT_HERSHEY_PLAIN, 1.3, (255, 255, 255), 2)
                    cv2.putText(img, f'HDist={int(horiz_dist)} Len: I={int(index_length)} M={int(middle_length)}', (20, 160),
                                cv2.FONT_HERSHEY_PLAIN, 1.3, (255, 255, 255), 2)

                    # UP: tip is above pip (smaller Y in image coords)
                    if yIndex < yIndex_pip and yMiddle < yMiddle_pip:
                        pyautogui.scroll(speed)
                        cv2.putText(img, f'SCROLL UP', (wCam//2 - 120, 100),
                                    cv2.FONT_HERSHEY_PLAIN, 2, (0, 255, 0), 3)
                        cv2.putText(img, f'Speed: {speed}', (wCam//2 - 80, 140),
                                    cv2.FONT_HERSHEY_PLAIN, 1.5, (0, 255, 0), 2)
                        cv2.line(img, (xIndex, yIndex), (xMiddle, yMiddle), (0,255,0), 3)
                        time.sleep(0.06)

                    # DOWN: tip is below pip (larger Y in image coords)
                    elif yIndex > yIndex_pip and yMiddle > yMiddle_pip:
                        pyautogui.scroll(-speed)
                        cv2.putText(img, f'SCROLL DOWN', (wCam//2 - 130, 100),
                                    cv2.FONT_HERSHEY_PLAIN, 2, (0, 0, 255), 3)
                        cv2.putText(img, f'Speed: {speed}', (wCam//2 - 80, 140),
                                    cv2.FONT_HERSHEY_PLAIN, 1.5, (0, 0, 255), 2)
                        cv2.line(img, (xIndex, yIndex), (xMiddle, yMiddle), (0,0,255), 3)
                        time.sleep(0.06)
                    else:
                        # Show why it's not triggering
                        cv2.putText(img, f'Tilt fingers UP or DOWN', (wCam//2 - 200, 100),
                                    cv2.FONT_HERSHEY_PLAIN, 1.5, (255, 255, 0), 2)

            # ------------------- MOUSE MOVEMENT (only index finger) -------------------
            elif fingers[1] == 1 and fingers[2] == 0:
                if drag_active:
                    autopy.mouse.toggle(autopy.mouse.Button.LEFT, False)
                    drag_active = False

                x3 = np.interp(xIndex, (frameR, wCam-frameR), (0, wScr))
                y3 = np.interp(yIndex, (frameR, hCam-frameR), (0, hScr))

                clocX = plocX + (x3 - plocX)/smoothening
                clocY = plocY + (y3 - plocY)/smoothening

                autopy.mouse.move(wScr - clocX, clocY)
                plocX, plocY = clocX, clocY

            # ------------------- RIGHT CLICK (Thumb + Index close) -------------------
            length_rc, _, _ = detector.findDistance(4, 8, img, draw=False)
            if length_rc < 30 and fingers[2] == 0:
                autopy.mouse.click(autopy.mouse.Button.RIGHT)
                cv2.putText(img, 'RIGHT CLICK', (wCam//2 - 100, 100),
                           cv2.FONT_HERSHEY_PLAIN, 2, (255, 0, 255), 3)
                time.sleep(0.25)

        # FPS
        cTime = time.time()
        fps = 1 / (cTime - pTime + 0.0001)
        pTime = cTime
        cv2.putText(img, f'FPS: {int(fps)}', (20,50),
                    cv2.FONT_HERSHEY_PLAIN, 2, (255,0,0), 2)

        cv2.imshow("Virtual Mouse", img)
        if cv2.waitKey(1) == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()


if __name__ == "__main__":
    main()
